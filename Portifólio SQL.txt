  Aqui temos dois códigos que tiveram que ser adaptados para atendimento das informações ao cliente, nos pedidos de compra/entradas de notas foram utilizados %% LIKE para puxar o nome do cliente, sendo essa a única forma de identificar o cliente, no sistema também não há link entre pedido de compra com a indústria e o pedido de venda, o recebimento também não é feito
por lote o que dificulta alguns rastreios.
  O estoque fica compartilhado na mesma filial, sendo de diversos clientes, o estoque que já chega vendido tive que criar uma simulação de estoque virtual para omitir estoques de outros clientes, usei da seguinte forma, estoque de fornecedor compartilhado entre clientes = entrada de notas do cliente + devolução do cliente - qtd faturada = estoque atual no CD, desta forma consegui criar um valor próximo a realidade para ser apresentada ao cliente no BI.





PEDIDO DE COMPRA COM A INDÚSTRIA

SELECT pcfornec.fornecedor,
       pcpedido.DTEMISSAO AS "DATA DE EMISSÃO",
       pcitem.NUMPED AS "Nº PEDIDO DISTRIB P IND",
       pcpedido.obs7 AS "REF PEDIDO CLI",
       SUM(pcitem.qtpedida/pcprodut.QTUNITCX) as "QTDCX DIGITADA",
       SUM(pcitem.qtentregue/pcprodut.QTUNITCX) as "QTDCX ENTREGUE",
       SUM(pcitem.QTPEDIDA * pcitem.PCOMPRA)+SUM(pcitem.qtpedida*pcitem.VLIPI) AS "VALOR DIGITADO",
       SUM(pcitem.QTENTREGUE * pcitem.PCOMPRA)+SUM(pcitem.qtentregue*pcitem.VLIPI) AS "VALOR ENTREGUE",
      SUM(pcitem.QTPEDIDA * (pcitem.PCOMPRA + pcitem.VLIPI))
- SUM(pcitem.QTENTREGUE * (pcitem.PCOMPRA + pcitem.VLIPI)) AS "VALOR SALDO"
FROM SCHEMA.pcitem
INNER JOIN SCHEMA.pcprodut ON pcprodut.codprod = pcitem.codprod
INNER JOIN SCHEMA.pcpedido ON pcpedido.NUMPED = pcitem.NUMPED
INNER JOIN SCHEMA.pcfornec ON pcfornec.codfornec = pcprodut.codfornec
WHERE pcpedido.DTEMISSAO trunc(sysdate) - 180           = Sysdate - 180 pois o cliente só necessita de visualização recente para tratativas, com um pequeno rastro do passado para reuniões
  AND (                                                   com as indístrias.
       pcpedido.OBS  LIKE '%NOME DO CLIENTE%' OR
       pcpedido.OBS2 LIKE '%NOME DO CLIENTE%' OR
       pcpedido.OBS3 LIKE '%NOME DO CLIENTE%' OR
       pcpedido.OBS4 LIKE '%NOME DO CLIENTE%' OR           = Nesta parte precisei colocar diversos like para buscar o nome do cliente no pedido de compra com a indústria,
       pcpedido.OBS5 LIKE '%NOME DO CLIENTE%' OR             devido a forma sem padrão a qual o processo o corre, precisei colocar like em todos os campos de obs. do pedido, para garantir
       pcpedido.OBS6 LIKE '%NOME DO CLIENTE%' OR             que apenas pedidos daquele cliente específico apareça no BI. (não há outro link para identificar a qual cliente aquela compra
       pcpedido.OBS7 LIKE '%NOME DO CLIENTE%'                 pertence)
      )
GROUP BY pcpedido.DTEMISSAO,
         pcitem.NUMPED,
         pcfornec.fornecedor,
         pcpedido.obs7



ESTOQUE

select CASE pcest.CODFILIAL
       WHEN '1' THEN 'DISTRIB_DISTRIB'          = IDENTIFICANDO TIPO DA OPERAÇÃO DA FILIAL ABERTA NO SISTEMA TOTVS WINTHOR
       WHEN '10' THEN 'DISTRIB_CROSS'
       END AS CODFILIALDISTRIB,
       pcfornec.FORNECEDOR,
       pcprodut.NBM as NCM,
       (select pcmov.DTMOV
       from SCHEMA.pcmov
       where pcmov.codprod = pcest.CODPROD
       and pcmov.CODOPER in ('E', 'EB')
       order by pcmov.NUMTRANSENT desc
       fetch first 1 row only) as "ULTIMA ENTRADA",
       (select pcmov.PTABELA
       from SCHEMA.pcmov                                           = SÉRIE DE SUBCONSULTAS QUE SERVEM PARA IDENTIFICAR DATAS DE ÚLTIMAS ENTRADAS DO PRODUTO, VALORES COM E SEM IPI DA
       where pcmov.CODPROD = pcest.CODPROD                             DA ÚLTIMA ENTRADA (SEM IPI É IMPORTANTE PARA O SIMULADOR DE PREÇOS)
       and pcmov.CODOPER = 'E'
       order by pcmov.NUMTRANSENT desc
       fetch first 1 row only) as "VALOR DE ENTRA S/ IPI",
       (select pcmov.punit
       from SCHEMA.pcmov
       where pcmov.CODPROD = pcest.CODPROD
       and pcmov.CODOPER = 'E'
       order by pcmov.NUMTRANSENT desc
       fetch first 1 row only) as "VALOR DE ENTRADA C/ IPI",
       pcprodut.CODAUXILIAR AS EAN,
       pcprodut.codprod AS "COD DISTRIB",
       pcprodut.DESCRICAO,
       GREATEST(0,
         LEAST(
           CASE WHEN pcfornec.CODFORNEC IN ('170', '172', '47', '668', '3230', '549')
           THEN
             NVL((SELECT SUM(pcmov.qt / NULLIF(pcmov.QTUNITCX, 0))
                  FROM SCHEMA.pcmov
                  INNER JOIN SCHEMA.pcnfent ON pcnfent.NUMTRANSENT = pcmov.NUMTRANSENT
                  INNER JOIN SCHEMA.pcpedido ON pcpedido.NUMPED = pcmov.NUMPED
                  WHERE pcmov.CODPROD = pcest.CODPROD
                  AND pcmov.CODOPER = 'E'
                  AND pcmov.NUMPED IS NOT NULL
                  AND pcmov.DTCANCEL IS NULL
                  AND pcnfent.DTENT > to_date('01/01/2026', 'dd/mm/yyyy')
                  AND (PCPEDIDO.OBS  LIKE '%DOM%' OR PCPEDIDO.OBS2 LIKE '%DOM%' OR
                       PCPEDIDO.OBS3 LIKE '%DOM%' OR PCPEDIDO.OBS4 LIKE '%DOM%' OR
                       PCPEDIDO.OBS5 LIKE '%DOM%' OR PCPEDIDO.OBS6 LIKE '%DOM%' OR                         = CASE QUE SIMULA UM ESTOQUE, COMO A DISTRIBUIDORA NÃO USA LOTE NOS SEUS
                       PCPEDIDO.OBS7 LIKE '%DOM%')),0)                                                       RECEBIMENTOS FIZ DA SEGUINTE FORMA DE PEGAR A ENTRADA REGISTRADA PARA AQUELE
             +                                                                                               CLIENTE, SOMAR COM DEVOLUÇÕES DAQUELE CLIENTE E SUBTRAIR PELAS QUANTIDADES
             NVL((SELECT SUM(pcmov.qt / NULLIF(pcmov.QTUNITCX, 0))                                           FATURADAS PARA AQUELE CLIENTE, GERANDO UM ESTOQUE VIRTUAL, OMITINDO OUTROS
                  FROM SCHEMA.pcmov                                                                        CLIENTES QUE COMPARTILHAM FORNECEDORES
                  INNER JOIN SCHEMA.pcclient ON pcclient.CODCLI = pcmov.CODCLI
                  WHERE pcmov.CODPROD = pcest.CODPROD
                  AND pcmov.CODOPER = 'ED'
                  AND pcmov.DTCANCEL IS NULL
                  AND pcmov.DTMOV > to_date('01/01/2026', 'dd/mm/yyyy')
                  AND pcclient.CLIENTE IN ('CLIENTE','CLIENTE')),0)
             -
             NVL((SELECT SUM(pcmov.qt / NULLIF(pcmov.QTUNITCX, 0))
                  FROM SCHEMA.pcmov
                  INNER JOIN SCHEMA.pcclient ON pcclient.CODCLI = pcmov.CODCLI
                  WHERE pcmov.CODPROD = pcest.CODPROD
                  AND pcmov.CODOPER = 'S'
                  AND pcmov.DTCANCEL IS NULL
                  AND pcmov.DTMOV > to_date('01/01/2026', 'dd/mm/yyyy')
                  AND pcclient.CLIENTE IN ('CLIENTE','CLIENTE')),0)
           ELSE
             (NVL(pcest.QTESTGER,0) - NVL(pcest.QTRESERV,0) - NVL(pcest.QTBLOQUEADA,0))
             / DECODE(NVL(pcprodut.QTUNITCX,0),0,1,pcprodut.QTUNITCX)
           END,
           GREATEST(0,
             (NVL(pcest.QTESTGER,0) - NVL(pcest.QTRESERV,0) - NVL(pcest.QTBLOQUEADA,0))                  = ESTOQUE SEM PEDIDO, USEI O ESTOQUE GERAL - O BLOQUEADO - O RESERVADO, JÁ QUE A
             / DECODE(NVL(pcprodut.QTUNITCX,0),0,1,pcprodut.QTUNITCX))                                     DISTRIBUIDORA TRABALHA COM ESTOQUE E PEDIDO EM DIVERSOS STATUS.
         )
       ) AS "QTD ESTOQUE DISPONIVEL",
       GREATEST(0, (NVL(pcest.QTBLOQUEADA,0) - NVL(pcest.QTINDENIZ,0)) / DECODE(NVL(pcprodut.QTUNITCX,0),0,1,pcprodut.QTUNITCX)) as "QTD ESTOQUE BLOQUEADO", = pcest.QTINDENIZ RESIDE DENTRO
       NVL(pcest.QTRESERV,0) / DECODE(NVL(pcprodut.QTUNITCX,0),0,1,pcprodut.QTUNITCX) as "QTD ESTOQUE RESERVADO",                                           DO BLOQUEADO, PRECISA SER
       NVL(pcest.QTINDENIZ,0) / DECODE(NVL(pcprodut.QTUNITCX,0),0,1,pcprodut.QTUNITCX) as "QTD ESTOQUE AVARIADO",                                           RETIRADA, POIS É AVARIA
       GREATEST(0,
         LEAST(
           CASE WHEN pcfornec.CODFORNEC IN ('170', '172', '47', '668', '3230', '549')
           THEN
             (NVL((SELECT SUM(pcmov.qt / NULLIF(pcmov.QTUNITCX, 0))
                  FROM SCHEMA.pcmov
                  INNER JOIN SCHEMA.pcnfent ON pcnfent.NUMTRANSENT = pcmov.NUMTRANSENT
                  INNER JOIN SCHEMA.pcpedido ON pcpedido.NUMPED = pcmov.NUMPED
                  WHERE pcmov.CODPROD = pcest.CODPROD
                  AND pcmov.CODOPER = 'E'
                  AND pcmov.NUMPED IS NOT NULL
                  AND pcmov.DTCANCEL IS NULL
                  AND pcnfent.DTENT > to_date('01/01/2026', 'dd/mm/yyyy')
                  AND (PCPEDIDO.OBS  LIKE '%DOM%' OR PCPEDIDO.OBS2 LIKE '%NOME DO CLIENTE%' OR
                       PCPEDIDO.OBS3 LIKE '%DOM%' OR PCPEDIDO.OBS4 LIKE '%NOME DO CLIENTE%' OR
                       PCPEDIDO.OBS5 LIKE '%DOM%' OR PCPEDIDO.OBS6 LIKE '%NOME DO CLIENTE%' OR
                       PCPEDIDO.OBS7 LIKE '%DOM%')),0)
             +
             NVL((SELECT SUM(pcmov.qt / NULLIF(pcmov.QTUNITCX, 0))
                  FROM SCHEMA.pcmov
                  INNER JOIN SCHEMA.pcclient ON pcclient.CODCLI = pcmov.CODCLI
                  WHERE pcmov.CODPROD = pcest.CODPROD
                  AND pcmov.CODOPER = 'ED'
                  AND pcmov.DTCANCEL IS NULL
                  AND pcmov.DTMOV > to_date('01/01/2026', 'dd/mm/yyyy')
                  AND pcclient.CLIENTE IN ('CLIENTE','CLIENTE')),0)
             -
             NVL((SELECT SUM(pcmov.qt / NULLIF(pcmov.QTUNITCX, 0))
                  FROM SCHEMA.pcmov
                  INNER JOIN SCHEMA.pcclient ON pcclient.CODCLI = pcmov.CODCLI
                  WHERE pcmov.CODPROD = pcest.CODPROD
                  AND pcmov.CODOPER = 'S'
                  AND pcmov.DTCANCEL IS NULL
                  AND pcmov.DTMOV > to_date('01/01/2026', 'dd/mm/yyyy')
                  AND pcclient.CLIENTE IN ('CLIENTE','CLIENTE')),0))
             * DECODE(NVL(pcprodut.QTUNITCX,0),0,1,pcprodut.QTUNITCX)
             * NVL((select pcmov.punit from SCHEMA.pcmov where pcmov.CODPROD = pcest.codprod
                    and pcmov.CODOPER = 'E'
                    order by pcmov.DTMOV desc, pcmov.numtransent desc
                    fetch first 1 row only),0)
           ELSE
             (NVL(pcest.QTESTGER,0) - NVL(pcest.QTRESERV,0) - NVL(pcest.QTBLOQUEADA,0))
             * NVL((select pcmov.punit from SCHEMA.pcmov where pcmov.CODPROD = pcest.codprod
                    and pcmov.CODOPER = 'E'
                    order by pcmov.DTMOV desc, pcmov.numtransent desc
                    fetch first 1 row only),0)
           END,
           GREATEST(0,
             (NVL(pcest.QTESTGER,0) - NVL(pcest.QTRESERV,0) - NVL(pcest.QTBLOQUEADA,0))
             * NVL((select pcmov.punit from SCHEMA.pcmov where pcmov.CODPROD = pcest.codprod
                    and pcmov.CODOPER = 'E'
                    order by pcmov.DTMOV desc, pcmov.numtransent desc
                    fetch first 1 row only),0))
         )
       ) as "VALOR EST DISPONIVEL",
       GREATEST(0, ((NVL(pcest.QTBLOQUEADA,0) - NVL(pcest.QTINDENIZ,0)) * NVL((select pcmov.punit from SCHEMA.pcmov where pcmov.CODPROD = pcest.codprod and pcmov.CODOPER = 'E' order by pcmov.DTMOV desc, pcmov.numtransent desc fetch first 1 row only),0))) as "VALOR EST BLOQUEADO",
       (NVL(pcest.QTRESERV,0) * NVL((select pcmov.punit from SCHEMA.pcmov where pcmov.CODPROD = pcest.codprod and pcmov.CODOPER = 'E' order by pcmov.DTMOV desc, pcmov.numtransent desc fetch first 1 row only),0)) as "VALOR EST RESERVADO",
       pcprodut.QTUNITCX
from agsimoes.pcest
inner join agsimoes.pcprodut on pcprodut.CODPROD = pcest.CODPROD
inner join agsimoes.pcfornec on pcfornec.CODFORNEC = pcprodut.CODFORNEC
where pcest.CODFILIAL = '1'
and pcprodut.revenda = 'S'
AND pcfornec.codfornec NOT IN ('76', '75', '3462', '3415', '3410', '3583', '3590', '3253', '3546', '3593', '3421', '3456', '3198',
    '3176', '3177', '3162', '3240', '3221', '2701', '2692', '2082', '2614','3487', '3505',                                              = EXCLUI ALGUNS FORNECEDORES E LIMITEI A PRODUTOS
    '3592', '3379', '3169', '3186', '592', '3427', '3453', '3614', '3226', '3227')                                                        MARCADOS COMO REVENDA
